function toCSV(rows, cols) {

  const h = cols.map(c => c.label).join(",");

  const b = rows.map(r => cols.map(c => `"${(r[c.key] ?? "").toString().replace(/"/g, '""')}"`).join(",")).join("\n");

  return h + "\n" + b;

}

function downloadCSV(content, filename) {

  const blob = new Blob(["\uFEFF" + content], { type: "text/csv;charset=utf-8;" });

  const url = URL.createObjectURL(blob);

  const a = document.createElement("a"); a.href = url; a.download = filename; a.click();

  URL.revokeObjectURL(url);

}

  async function saveConfig(key, arr) {

    const rows = await sbGet("config", `clave=eq.${key}&select=id`);

    if (rows.length) await sbPatch("config", rows[0].id, { valor: arr });

    else await sbPost("config", { clave: key, valor: arr });

    setConfigState(prev => ({ ...prev, [key]: arr }));

  }

  async function saveClient(form) {

    setSaving(true);

    try {

      const body = { nombre: form.nombre, nif: form.nif, calle: form.calle, numero: form.numero, cp: form.cp, ciudad: form.ciudad, pais: form.pais, telefono: form.telefono, email: form.email, contacto: form.contacto, canal: form.canal, grupo: form.grupo, gestor: form.gestor };

      if (form.id) await sbPatch("clientes", form.id, body);

      else await sbPost("clientes", body);

      await loadAll(); setClientPanel(null);

    } catch (e) { alert("Error al guardar: " + e.message); }

    finally { setSaving(false); }

  }

  async function deleteClient(id) {

    if (!confirm("¿Eliminar este cliente y todos sus datos?")) return;

    try { await sbDelete("clientes", id); await loadAll(); setDetalleId(null); }

    catch (e) { alert("Error: " + e.message); }

  }

  async function toggleTask(t) {

    catch (e) { alert("Error: " + e.message); }

  }

          <div style={{ fontSize: 12, color: "#888", marginTop: 2 }}>Gestión de clientes B2B</div>

        </div>

        <div style={{ display: "flex", alignItems: "center", gap: 8 }}>

          <div style={{ width: 9, height: 9, borderRadius: "50%", background: loading ? "#ba7517" : error ? "#e24b4a" : "#1d9e75" }} />

          <span style={{ fontSize: 12, color: "#888" }}>{loading ? "Conectando…" : error ? "Sin conexión" : "Conectado · Supabase"}</span>

        </div>

      </div>

      {/* TABS */}

      <div style={{ display: "flex", gap: 2, marginBottom: 24, borderBottom: "1px solid #e8e8e8", overflowX: "auto" }}>

        {TABS.map(t => (

          <button key={t} onClick={() => setTab(t)}

            style={{ padding: "8px 16px", background: "none", border: "none", cursor: "pointer", fontSize: 13, fontWeight: tab === t ? 600 : 400, color: tab === t ? "#185fa5" : "#666", borderBottom: tab === t ? "2px solid #185fa5" : "2px solid transparent", marginBottom: -1, whiteSpace: "nowrap" }}>{t}</button>

        ))}

      </div>

      {/* CONTENT */}

      {loading ? <Spinner /> : error ? <ErrMsg msg={error} onRetry={loadAll} /> : <>

        {tab === "Dashboard"     && <Dashboard />}

        {tab === "Clientes"      && <TabClientes />}

        {tab === "Actividades"   && <TabActividades />}

        {tab === "Tareas"        && <TabTareas />}

        {tab === "Informes"      && <TabInformes />}

        {tab === "Configuración" && <TabConfiguracion />}

      </>}

    </div>

  );

}
