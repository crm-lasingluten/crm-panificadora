function toCSV(rows, cols) {
  const h = cols.map(c => c.label).join(",");
  const b = rows.map(r => cols.map(c => `"${(r[c.key] ?? "").toString().replace(/"/g, '""')}"`).join(",")).join("\n");
  return h + "\n" + b;
}
function downloadCSV(content, filename) {
  const blob = new Blob(["\uFEFF" + content], { type: "text/csv;charset=utf-8;" });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a"); a.href = url; a.download = filename; a.click();
  URL.revokeObjectURL(url);
}

    if (!confirm("¿Eliminar este cliente y todos sus datos?")) return;

  async function saveActivity(form) {
    setSaving(true);
    try {
      const body = { cliente_id: form.clienteId, tipo: form.tipo, nota: form.nota, fecha: form.fecha, gestor: form.gestor };
      if (form.id) await sbPatch("actividades", form.id, body);
      else await sbPost("actividades", body);
      await loadAll(); setActPanel(null);
    } catch (e) { alert("Error: " + e.message); }
    finally { setSaving(false); }
  }
  async function deleteActivity(id) {
    try { await sbDelete("actividades", id); setActivities(p => p.filter(a => a.id !== id)); }
    catch (e) { alert("Error: " + e.message); }
  }
  async function saveTask(form) {
    setSaving(true);
    try {
      const body = { cliente_id: form.clienteId, titulo: form.titulo, fecha: form.fecha, prioridad: form.prioridad, estado: form.estado, gestor: form.gestor };
      if (form.id) await sbPatch("tareas", form.id, body);
      else await sbPost("tareas", body);
      await loadAll(); setTaskPanel(null);
    } catch (e) { alert("Error: " + e.message); }
    finally { setSaving(false); }
  }
  async function toggleTask(t) {
    const estado = t.estado === "Completada" ? "Pendiente" : "Completada";
    try { await sbPatch("tareas", t.id, { estado }); setTasks(p => p.map(x => x.id === t.id ? { ...x, estado } : x)); }
    catch (e) { alert("Error: " + e.message); }
  }
  async function deleteTask(id) {
    try { await sbDelete("tareas", id); setTasks(p => p.filter(t => t.id !== id)); }
    catch (e) { alert("Error: " + e.message); }
  }
